CREATE PROCEDURE add_user(IN un VARCHAR(30),IN ue TEXT,IN fn VARCHAR(30),IN ln VARCHAR(30),IN up TEXT,IN g VARCHAR(30))
BEGIN
INSERT INTO users (user_name,user_email,first_name,last_name,user_password,gender) VALUES (un,ue,fn,ln,up,g);
END

CREATE PROCEDURE insert_notes(IN nt VARCHAR(100), IN nd VARCHAR (500), IN a INT, IN nf TEXT, IN ns INT, IN ud DATE)
BEGIN
INSERT INTO notes (notes_title, notes_description, author, notes_file, notes_subject, upload_date) VALUES (nt,nd,a,nf,ns,ud);
END

CREATE PROCEDURE update_notes(IN nt VARCHAR(100),IN nd VARCHAR(500), IN ns INT, IN ni INT)
BEGIN
UPDATE notes SET notes_title=nt, notes_description=nd, notes_subject=ns WHERE notes_id=ni;
END;

CREATE PROCEDURE validate_login(IN un VARCHAR(30),IN up TEXT)
BEGIN
SELECT u.user_id, u.user_name, ut.user_type FROM users u JOIN user_type ut ON u.user_type=ut.user_type_id WHERE (u.user_name=un OR u.user_email=un) AND u.user_password=up;
END

CREATE PROCEDURE select_notes(IN ui INT)
BEGIN
SELECT * FROM notes WHERE author=ui;
END

CREATE PROCEDURE counts(IN ui INT)
BEGIN
SELECT COUNT(*) AS following FROM followers WHERE following=ui;
SELECT COUNT(*) AS followers FROM followers WHERE follower=ui;
SELECT COUNT(*) AS notes FROM notes WHERE author=ui;
END

BEGIN
    SELECT COUNT(*) AS following FROM followers WHERE following=ui UNION SELECT COUNT(*) AS followers FROM followers WHERE follower=ui UNION SELECT COUNT(*) AS notes FROM notes WHERE author=ui;
